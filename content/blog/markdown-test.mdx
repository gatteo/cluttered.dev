function cleanNodeModules(projectPath) {
  const nodeModules = path.join(projectPath, 'node_modules');

  if (fs.existsSync(nodeModules)) {
    console.log(`Cleaning: ${nodeModules}`);
    fs.rmSync(nodeModules, { recursive: true });
    return true;
  }

  return false;
}
